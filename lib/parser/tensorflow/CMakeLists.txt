set(NAME TFPARSER)

# Source files.
set(SRCS
  tf_parser.cc
)

set(LLVM_TARGET_DEFINITIONS
  ${CMAKE_SOURCE_DIR}/include/halo/lib/ir/tf_convert.td
)

halo_tblgen(tf_convert.h.inc -gen-converter-decl)
halo_tblgen(tf_convert.cc.inc -gen-converter-def)
halo_tblgen(tf_regist_op.h.inc -gen-register-op)
add_public_tablegen_target(TF_GEN)

# Dependences which need to be built first.
set(DEPENDENCES
  PARSER
  TF_GEN
)

create_halo_object(TARGET_NAME ${NAME}
  TARGET_SRCS ${SRCS}
  TARGET_DEPENDENCES ${DEPENDENCES}
)

target_include_directories(${NAME} PRIVATE
  ${LLVM_SRC_DIR}/include
  ${CMAKE_BINARY_DIR}/llvm/include
)

gen_protobuf_files(TARGET_NAME ${NAME}
  PROTO_DIR ${CMAKE_SOURCE_DIR}/external/protos/tensorflow
)
